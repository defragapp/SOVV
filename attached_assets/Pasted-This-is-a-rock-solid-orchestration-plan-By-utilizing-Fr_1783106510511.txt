### 1. The Orchestration Hook (`useHeroEntrance.ts`)

Create this hook to manage the timeline, respect `prefers-reduced-motion`, and handle the one-time `sessionStorage` flag.

```typescript
import { useEffect, useState } from 'react';
import { useAnimate, useReducedMotion } from 'framer-motion';

export function useHeroEntrance() {
  const [scope, animate] = useAnimate();
  const shouldReduceMotion = useReducedMotion();
  const [isReady, setIsReady] = useState(false);

  useEffect(() => {
    const hasPlayed = sessionStorage.getItem('hero-entrance-played');
    
    // Fallback for reduced motion or returning visitors
    if (hasPlayed || shouldReduceMotion) {
      setIsReady(true);
      return;
    }

    const playCinematicEntrance = async () => {
      // The single continuous ease curve for grounded structural motion
      const cinematicEase = [0.16, 1, 0.3, 1];

      // 1. Ambient glow ignition (Spark)
      await animate(
        ".ambient-glow", 
        { scale: [0, 1], opacity: [0, 1] }, 
        { duration: 0.6, ease: "easeOut" }
      );

      // 2. Image wipe & Nav descend
      // Start nav descent in parallel, slightly offset
      animate(
        ".site-shell-nav", 
        { y: ["-100%", "0%"], opacity: [0, 1] }, 
        { duration: 0.5, delay: 0.3, ease: cinematicEase }
      );
      
      await animate(
        ".hero-image-container", 
        { clipPath: ["inset(0 100% 0 0)", "inset(0 0% 0 0)"] }, 
        { duration: 0.7, ease: cinematicEase }
      );

      // 3. Mask-reveal headline lines
      await animate(
        ".headline-line-1", 
        { y: ["100%", "0%"] }, 
        { duration: 0.6, ease: cinematicEase }
      );
      await animate(
        ".headline-line-2", 
        { y: ["100%", "0%"] }, 
        { duration: 0.6, ease: cinematicEase, delay: -0.2 } // Staggered overlap
      );

      // 4. Subtext and CTA materialize
      await animate(
        ".hero-subtext, .hero-cta", 
        { opacity: [0, 1], y: [10, 0] }, 
        { duration: 0.5, ease: "easeOut" }
      );

      sessionStorage.setItem('hero-entrance-played', 'true');
      setIsReady(true);
    };

    playCinematicEntrance();
  }, [animate, shouldReduceMotion]);

  return { scope, isReady };
}

```

### 2. Refactoring `HomePage.tsx`

Wrap the hero content in the animation scope. We are ensuring the headline lines have the `overflow-hidden` wrappers for that classic Fraunces display sweep.

```tsx
import { useHeroEntrance } from '@/hooks/useHeroEntrance'; // Adjust path as needed
import { SiteShell } from '@/components/marketing/site-shell';

export default function HomePage() {
  const { scope, isReady } = useHeroEntrance();

  // If the animation is skipped or finished, we remove the starting inline styles 
  // to prevent CSS conflicts on resize/interaction.
  const motionSafeClass = isReady ? "entrance-complete" : "entrance-pending";

  return (
    <div ref={scope} className={`relative min-h-screen ${motionSafeClass}`}>
      {/* Pass a flag to SiteShell so it knows whether to render in its 
        final state instantly, or wait for the orchestrator.
      */}
      <SiteShell isEntranceComplete={isReady} />

      <main className="relative flex flex-col items-center justify-center pt-32 pb-16">
        
        {/* Ambient Glow Ignition */}
        <div 
          className="ambient-glow absolute right-1/4 top-1/4 h-96 w-96 rounded-full bg-amber-500/20 blur-[100px]"

        {/* Hero Image Container */}
        <div 
          className="hero-image-container relative z-0 w-full max-w-5xl"
```
